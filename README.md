# git

### check for changes

git status

### add for staging

git add filename.js

### commit change

git commit -m 'commit comment'

### amend commit (minor change)

git commit --amend

### remove accidentally commited commit

git rm —-cached file-to-remove.js

git commit —-amend

### sync branch with main

git checkout branchName

git pull origin mainBranchName
